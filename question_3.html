#Universal function we used to determine if a song is 'most streamed'
#Returns 1 if the song is in the top 10% of streams, 0 otherwise
def top_10_percent(streams):
    threshold = spotify_data['Stream'].quantile(0.9)
    return 1 if streams >= threshold else 0
In[]:
#Apply the function to the dataframe to create a new column

#import needed libraries for sentiment analysis
import nltk
In[]:
#Download the sentiment analyzer that will be used and intializing a variable that llows access
#Create a function that will return the sentiment of a string passed
from nltk.sentiment.vader import SentimentIntensityAnalyzer
sent_analyzer = SentimentIntensityAnalyzer()
def sentiment_scores(docx):
    return sent_analyzer.polarity_scores(docx.text)
In[]:
#import spacy and load the english model
import spacy
In[]:
nlp = spacy.load('en_core_web_sm')
In[]:
#Now using the intialized variable, we can prep the sentiment analyzer for the dataframe
from spacy.tokens import Doc
Doc.set_extension("sentimenter",getter=sentiment_scores, force=True)
In[]:
#Access the compound score of the sentiment analyzer and create a column that holds the score
most_streamed['Track_sentiment'] = most_streamed['Track'].apply(lambda x: nlp(x)._.sentimenter['compound'])
